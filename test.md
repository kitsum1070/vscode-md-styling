fn part1(contents: &str) {
    // mul(x,y)
    let re = Regex::new(r"mul\(([0-9]{1,3})+,([0-9]{1,3})\)").unwrap();
    println!("DAY03 Part 1: {}", re.captures_iter(contents).map(|cap| {
        let x: i32 = cap[1].parse().unwrap();
        let y: i32 = cap[2].parse().unwrap();
        x * y
    }).sum::<i32>());
}
